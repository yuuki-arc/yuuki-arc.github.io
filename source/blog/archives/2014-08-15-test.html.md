* SQL構文

```sql
select distinct c.aa_code, u.aa_text,c.begin_date
from container c
inner join billding b
   on c.container_id = b.container_id
inner join matchers u
   on c.aa_code = u.aa_code
 left outer join selecter r
   on c.aa_code = r.aa_code
 left outer join register r2
   on c.aa_code = r2.aa_code
where b.state <> ''
  and b.sub = 'ii'
  and r.aa_code is null
  and r2.aa_code is null
order by c.begin_date, c.aa_code
;
```
